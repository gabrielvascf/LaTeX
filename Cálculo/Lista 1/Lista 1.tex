\soln{Problema 2: }
\setcounter{SubProblemNum}{0}
\(h = 10 + 120t - 5t^2\)
\part{
	A altura do foguete 2 segundos após ser lançado:
	\begin{gather*}
		h = 10 + 120t - 5t^2    \\
		h = 10 + 120*2 - 5(2^2) \\
		h = 10 + 240 - 5*4      \\
		h = 250 - 20            \\
		h = 230 \text{ metros}
	\end{gather*}
}
\part{
	O tempo necessário para o foguete atingir a altura de 485 metros:
	\begin{gather*}
		h = 10 + 120t - 5t^2    \\
		485 = 10 + 120t - 5t^2 \\
		5t^2 + 120t - 475 = 0\\
	\end{gather*}
	Usando a fórmula de báskara:
	\begin{equation*}
		S = \frac{-b \pm \sqrt{\Delta}}{2a} \\
	\end{equation*}
	\begin{gather*}
		\begin{split}
			\Delta & = b^2-4ac            \\
			       & = 120^2 - 4*10*-475  \\
			       & = 1440 - (40*-19000) \\
			       & = 1440 + 19000       \\
			       & = 20440              \\
		\end{split}
		\\
		\begin{split}
			S & = \frac{-120 \pm \sqrt{20440}}{20} \\
			  & = 5, 19
		\end{split}
	\end{gather*}
	\emph{5 segundos.}
}
\soln{Problema 3: }
\setcounter{SubProblemNum}{0}
\part{
	Encontre a função lucro $L$, sendo que o lucro é igual à Receita menos Custo, ou seja, $L(d) = R(d) - C(d)$.
	\begin{equation*}
		R(d) = -d^2 + 31d-30
	\end{equation*}
	\begin{equation*}
		C(d) = 11d - 19
	\end{equation*}
	\begin{equation} \label{lucro}
		\begin{split}
			L(d) & = (-d^2 + 31d - 30) - (11d - 19) \\
			     & = -d^2 + 31d - 30 - 11d - 19     \\
			     & = -d^2 + 21d - 49                \\
		\end{split}
	\end{equation}
}
\part{
	Em que dias o lucro da empresa é zero?
	Usando a equação para lucro (\ref{lucro}), podemos calcular:
	\begin{multline*}
		\\L(x) = 0                                              \\
		-d^2 + 21d - 49 = 0                                   \\
		\text{Usando a fórmula de báskara: }                  \\
		S = \frac{-21 \pm \sqrt{21^2 - 4 * -1 * -49}}{2*-1} \\
		S = \frac{-21 \pm \sqrt{441 - 196}}{-2}             \\
		S = \frac{-21 \pm \sqrt{245}}{-2} \\
		S = \frac{-21 + \sqrt{245}}{-2}, \frac{-21 - \sqrt{245}}{-2} \\
		S = 2.6737\dots, 18.3262\dots
	\end{multline*}
	\emph{Aproximadamente nos dias 3 e 18 de cada mês.}
}

	Problema 4: O saldo de uma conta bancária é dado por $S = t^2 - 11t+24$ onde $S$ é o saldo em reais e $t$ é o tempo em dias. Determine:
}
\setcounter{SubProblemNum}{0}
\part{

	\begin{multline*}
		\\S(t) = t^2 - 11t + 24 \\
		\text{Usando a fórmula da báskara: }\\
		S = \frac{11 \pm \sqrt{-11^2 - 4 * 1 * 24}}{2*1}\\
		S = \frac{11 \pm \sqrt{121 - 96}}{2} \\
		S = \frac{11 \pm \sqrt{25}}{2} \\
		S = \frac{11 \pm 5}{2} \\
		S = \frac{11 + 5}{2}, \frac{11 - 5}{2} \\
		S = \frac{16}{2}, \frac{6}{2} \\
		S = 8, 3
	\end{multline*}

}

\part{
	Em que dia o saldo é mínimo:
	\begin{multline*}
		\text{Pela fórmula do vértice da parábola:}
